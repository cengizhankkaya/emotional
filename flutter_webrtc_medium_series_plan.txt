```dart
class SimpleCameraScreen extends StatefulWidget {

class _SimpleCameraScreenState extends State<SimpleCameraScreen> {

    initRenderers();
    getUserMedia();

      'audio': true,
      'video': {
        'facingMode': 'user',
        'width': 1280,
        'height': 720,
      }
    };

    try {

          .getUserMedia(mediaConstraints);
      _localRenderer.srcObject = stream;
      setState(() {
        _localStream = stream;
      });
    } catch (e) {

    }
  }

    return Scaffold(
      appBar: AppBar(title: Text('WebRTC Kamera')),
      body: RTCVideoView(_localRenderer, mirror: true),
    );

    _localRenderer.dispose();
    super.dispose();
  }
```

#### 4. Firebase ile Signaling (2500 kelime)

**Signaling Service**:
```dart
class SignalingService {
  final FirebaseDatabase _database;
  
  // Offer gönder
  Future<void> sendOffer(String roomId, RTCSessionDescription sdp) async {
    await _database.ref('rooms/$roomId/offer').set({
      'type': sdp.type,
      'sdp': sdp.sdp,
    });
  }
  
  // Offer dinle
  Stream<RTCSessionDescription> listenForOffer(String roomId) {
    return _database.ref('rooms/$roomId/offer').onValue.map((event) {
      final data = event.snapshot.value as Map?;
      return RTCSessionDescription(
        data?['sdp'],
        data?['type'],
      );
    });
  }
  
  // ICE candidate gönder
  Future<void> sendIceCandidate(
    String roomId,
    String peerId,
    RTCIceCandidate candidate,
  ) async {
    await _database
        .ref('rooms/$roomId/candidates/$peerId')
        .push()
        .set({
      'candidate': candidate.candidate,
      'sdpMid': candidate.sdpMid,
      'sdpMLineIndex': candidate.sdpMLineIndex,
    });
  }
  
  // ICE candidates dinle
  Stream<RTCIceCandidate> listenForIceCandidates(
    String roomId,
    String peerId,
  ) {
    return _database
        .ref('rooms/$roomId/candidates/$peerId')
        .onChildAdded
        .map((event) {
      final data = event.snapshot.value as Map;
      return RTCIceCandidate(
        data['candidate'],
        data['sdpMid'],
        data['sdpMLineIndex'],
      );
    });
  }
}
```

#### 5. RTCPeerConnection Yaşam Döngüsü (2000 kelime)

**Tam Implementation**:
```dart
class WebRTCManager {
  RTCPeerConnection? _peerConnection;
  MediaStream? _localStream;
  final SignalingService _signaling;
  
  Future<void> createPeerConnection() async {
    final configuration = {
      'iceServers': [
        {'urls': 'stun:stun.l.google.com:19302'}
      ]
    };
    
    _peerConnection = await createPeerConnection(configuration);
    
    // Event handlers
    _peerConnection!.onIceCandidate = (candidate) {
      _signaling.sendIceCandidate(roomId, peerId, candidate);
    };
    
    _peerConnection!.onIceConnectionState = (state) {
      print('ICE Connection State: $state');
    };
    
    _peerConnection!.onAddStream = (stream) {
      _remoteRenderer.srcObject = stream;
    };
    
    // Local stream ekle
    _localStream!.getTracks().forEach((track) {
      _peerConnection!.addTrack(track, _localStream!);
    });
  }
  
  Future<void> makeCall(String roomId) async {
    // Offer oluştur
    RTCSessionDescription offer = await _peerConnection!.createOffer();
    await _peerConnection!.setLocalDescription(offer);
    await _signaling.sendOffer(roomId, offer);
    
    // Answer dinle
    _signaling.listenForAnswer(roomId).listen((answer) async {
      await _peerConnection!.setRemoteDescription(answer);
    });
    
    // ICE candidates dinle
    _signaling.listenForIceCandidates(roomId, peerId).listen((candidate) {
      _peerConnection!.addCandidate(candidate);
    });
  }
  
  Future<void> answerCall(String roomId) async {
    // Offer dinle
    _signaling.listenForOffer(roomId).listen((offer) async {
      await _peerConnection!.setRemoteDescription(offer);
      
      // Answer oluştur
      RTCSessionDescription answer = await _peerConnection!.createAnswer();
      await _peerConnection!.setLocalDescription(answer);
      await _signaling.sendAnswer(roomId, answer);
    });
    
    // ICE candidates dinle
    _signaling.listenForIceCandidates(roomId, peerId).listen((candidate) {
      _peerConnection!.addCandidate(candidate);
    });
  }
}
```

```dart
class DataChannelChatService {

  void sendTextMessage(String message) {
    _channel?.send(RTCDataChannelMessage(message));
  }
  
  void sendJson(Map<String, dynamic> data) {
    final json = jsonEncode(data);
    _channel?.send(RTCDataChannelMessage(json));
  }

    _channel?.send(RTCDataChannelMessage.fromBinary(data));
  }
  
  Future<void> sendFile(File file) async {
    final bytes = await file.readAsBytes();
    final chunkSize = 16384; // 16KB chunks
    
    for (var i = 0; i < bytes.length; i += chunkSize) {
      final end = (i + chunkSize < bytes.length) 
          ? i + chunkSize 
          : bytes.length;
      final chunk = bytes.sublist(i, end);
      _channel?.send(RTCDataChannelMessage.fromBinary(chunk));
    }
  }
}
```

## 📚 Part 5: Grup Görüşmeleri ve SFU

**Hedef**: 3+ kişili grup görüşmeleri implementasyonu

### İçerik Yapısı

#### 1. Mesh vs SFU vs MCU (1500 kelime)
- **Mesh Topology**
  - Her peer birbiriyle bağlanır
  - N*(N-1)/2 connection
  - Bandwidth intensive
  - 2-4 kişi için ideal

- **SFU (Selective Forwarding Unit)**
  - Media server forwards streams
  - N connection per client
  - Scalable (100+ participants)
  - Recommended

- **MCU (Multipoint Control Unit)**
  - Server mixes all streams
  - 1 connection per client
  - High server CPU
  - Legacy approach

#### 2. Mesh Implementation (2000 kelime)
- Multiple peer connections
- Connection management
- Stream synchronization

```dart
class MeshCallManager {
  Map<String, RTCPeerConnection> _peers = {};
  
  Future<void> addPeer(String userId) async {
    final pc = await createPeerConnection(config);
    
    // Local stream tracks ekle
    _localStream!.getTracks().forEach((track) {
      pc.addTrack(track, _localStream!);
    });
    
    // Event handlers
    pc.onTrack = (event) {
      if (event.streams.isNotEmpty) {
        _remoteStreams[userId] = event.streams[0];
      }
    };
    
    _peers[userId] = pc;
  }
  
  Future<void> removePeer(String userId) async {
    await _peers[userId]?.close();
    _peers.remove(userId);
    _remoteStreams.remove(userId);
  }
}
```
